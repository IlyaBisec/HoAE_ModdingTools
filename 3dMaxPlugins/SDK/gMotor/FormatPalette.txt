////////////////////////////////////////////////////////////////////////////////
// Palette block format
////////////////////////////////////////////////////////////////////////////////
	Header {
		DWORD	Magic;				// "GPAL"
		DWORD	FileSize;			// total size of the file
		DWORD	UnpackedSize;		// size for unpack
		WORD	Flags;				// flags (pack method)
		WORD	Reserved;			// reserved (zero)
	};
	BYTE PackedData[];
////////////////////////////////////////////////////////////////////////////////
// Unpacked data format
////////////////////////////////////////////////////////////////////////////////
	WORD PalettesNumber;	// number of palettes
	Palette {
		DWORD SizeOf;		// total size of the palette in bytes
		WORD 	Items;		// number of items in the palette
		WORD	Flags;		// flags
		BYTE* Data[];
	} Palettes[];
