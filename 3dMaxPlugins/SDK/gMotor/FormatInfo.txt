////////////////////////////////////////////////////////////////////////////////
// Info block format
////////////////////////////////////////////////////////////////////////////////
	Header {
		DWORD	Magic;				// "GINF"
		DWORD	FileSize;			// total size of the file
		DWORD	UnpackedSize;		// size for unpack
		WORD	Flags;				// flags (pack method)
		WORD	Reserved;			// reserved (zero)
	};
	BYTE PackedData[];
////////////////////////////////////////////////////////////////////////////////
// Unpacked data format
////////////////////////////////////////////////////////////////////////////////
	WORD ItemsNumber;		// number of items
	Item {
		DWORD SizeOf;		// total size of the item in bytes
		WORD	Flags;		// flags
		BYTE* Data[];
	} Items[];	
////////////////////////////////////////////////////////////////////////////////
